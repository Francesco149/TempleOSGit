GRA graphics files are 8-bits-per-pixels but only 4-bits of color, with 
transparency and no palette.  Compression is the standard TempleOS LZW 
compression.

#define DCF_COMPRESSED		1 //This is the only saved flag.

#define TRANSPARENT		0xFF
#define BLACK			0
#define BLUE			1
#define GREEN			2
#define CYAN			3
#define RED			4
#define PURPLE			5
#define BROWN			6
#define LTGRAY			7
#define DKGRAY			8
#define LTBLUE			9
#define LTGREEN			10
#define LTCYAN			11
#define LTRED			12
#define LTPURPLE		13
#define YELLOW			14
#define WHITE			15

class GRAFile
{
  I32	width;
  I32   width_internal; //Rounded-up to multiple of 8.
  I32   height;
  I32	flags;		//DCF_COMPRESSED? See [C:/Kernel/Compress.CPP.Z,1].
  U8	body[];
};
